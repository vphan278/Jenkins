pipeline {
  agent any

  stages {
    stage('Build & Test') {
      steps {
        // run Maven against the POM inside the BankAccount-Java folder
        sh 'mvn -B -f BankAccount-Java/pom.xml -DskipTests=false clean test'
      }
      post {
        always {
          // publish JUnit results
          junit 'BankAccount-Java/target/surefire-reports/*.xml'
        }
      }
    }

    stage('Package') {
      steps {
        sh 'mvn -B -f BankAccount-Java/pom.xml -DskipTests package'
      }
    }

    stage('Archive Artifact') {
      steps {
        archiveArtifacts artifacts: 'BankAccount-Java/target/*.jar', fingerprint: true
      }
    }
  }
}
