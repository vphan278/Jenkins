pipeline {
  agent any
  stages {
    stage('Build & Test') {
      steps {
        dir('BankAccount-Java') {
          sh 'mvn -B -DskipTests=false clean test'
        }
      }
      post {
        always {
          junit 'BankAccount-Java/target/surefire-reports/*.xml'
        }
      }
    }
    stage('Package') {
      steps {
        dir('BankAccount-Java') {
          sh 'mvn -B -DskipTests package'
        }
      }
    }
    stage('Archive Artifact') {
      steps {
        archiveArtifacts artifacts: 'BankAccount-Java/target/*.jar', fingerprint: true
      }
    }
  }
}
