pipeline {
  agent any
  tools {
    jdk 'JDK 17'        // name from Manage Jenkins → Tools (or remove if using system Java)
    maven 'maven3'      // name from Manage Jenkins → Tools
  }
  options { skipDefaultCheckout(true) }
  stages {
    stage('Checkout') {
      steps {
        checkout([$class: 'GitSCM',
          userRemoteConfigs: [[url: 'https://github.com/<you>/<repo>.git']],
          branches: [[name: '*/main']]
        ])
      }
    }
    stage('Build & Test') {
      steps {
        sh 'mvn -B -DskipTests=false clean test'
      }
      post {
        always {
          junit 'target/surefire-reports/*.xml'
        }
      }
    }
    stage('Package') {
      steps {
        sh 'mvn -B -DskipTests package'
      }
    }
    stage('Archive') {
      steps {
        archiveArtifacts artifacts: 'target/*.jar', fingerprint: true
      }
    }
  }
}
